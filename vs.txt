#define _CRT_SECURE_NO_WARNINGS 
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <math.h>
#include <ctype.h>
#define SIZE 1000

FILE* f;
FILE* g;


int push_char(char stack[], unsigned cur, char symb)
{
	if (cur < SIZE)
	{
		stack[cur++] = symb;
	}
	return cur;
}

int pop_char(char stack[], unsigned cur)
{
	if (cur != 0)
	{
		--cur;
	}
	return cur;
}

int is_empty_char(const char stack[], unsigned cur)
{
	return !cur;
}

f = fopen("input.txt", "rt");

}
